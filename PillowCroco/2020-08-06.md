#include <iostream>
#include <vector>
#include <queue>
using namespace std;

class node {
public:
	int time;
	int cum_time;
	vector<int> adj;
	int indeg;

	node(int t) : time(t) {
		cum_time = 0;
		adj = vector<int>();
		indeg = 0;
	}
};

int acm_craft(int n, vector<node> graph, int target){
	queue<node> Q;

	for (auto v : graph) {
		if (v.indeg == 0)
			Q.push(v);
	}

	while (!Q.empty()) {
		node cur = Q.front();
		Q.pop();
		for (auto v : cur.adj) {
			graph[v].cum_time = max(graph[v].cum_time, cur.cum_time + cur.time);
			graph[v].indeg--;
			if (graph[v].indeg == 0)
				Q.push(graph[v]);
		}

	}
	return graph[target].cum_time + graph[target].time;
}
