### 11722. 가장 긴 감소하는 부분 수열
(주소)https://www.acmicpc.net/problem/11722

#### 풀이 해설:

자기보다 앞에 있는 것들 중에서 / 자기보다 큰 원소들만 골라 / 그중에서 dp값이 가장 큰 애의 dp값에 +1 해주기

```c++
#include <iostream>
#include <algorithm>
#include <stack>
#include <vector>
using namespace std;
int N, A[1001], dp[1001];
int mm=0, mx=0;

int main(){
	cin>>N;
    for(int i=0;i<N;i++){
        cin>>A[i];
    }
    
    for(int i=0;i<N;i++){
        dp[i]=1; // 초기값
        mx=0;
        for(int j=0;j<i;j++){
            if(A[i]<A[j]){
                mx=max(mx, dp[j]);
                dp[i]=mx+1;
            }
        }
    }
    for(int i=0;i<N;i++){
        if(mm<dp[i]) mm=dp[i];
    }
    cout<<mm;
}
```

## Gold V

### 14499. 주사위 굴리기
(주소)https://www.acmicpc.net/problem/14499

#### 풀이 해설:

모든 순간마다 주사위 면 위치 ( 1 ~ 6 )는 무조건 예시와 동일하게 통일.

```c++
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int N,M,x,y,K,a;
vector<int> v; // 움직일 방향 순서 저장
int map[21][21];
int dice[7];  // 주사위 면에 적혀있는 수  : 인덱스 1 ~ 6만 사용. ( 각 숫자 자리는 문제 그림 참고 )
int dir[4][2]={{0,1},{0,-1},{-1,0},{1,0}}; // 동 서 북 남 : 1 2 3 4

void move(int d){
    int d1 = dice[1];
    int d2 = dice[2];
    int d3 = dice[3];
    int d4 = dice[4];
    int d5 = dice[5];
    int d6 = dice[6];
    if(d==1){
        dice[1]=d4;
        dice[3]=d1;
        dice[4]=d6;
        dice[6]=d3;
    }
    else if(d==2){
        dice[1]=d3;
        dice[3]=d6;
        dice[4]=d1;
        dice[6]=d4;
    }
    else if(d==3){
        dice[1]=d5;
        dice[2]=d1;
        dice[5]=d6;
        dice[6]=d2;
    }
    else if(d==4){
        dice[1]=d2;
        dice[2]=d6;
        dice[5]=d1;
        dice[6]=d5;
    }
}

int main(){
	cin>>N>>M>>x>>y>>K;
    for(int i=0;i<N;i++){
        for(int j=0;j<M;j++){
            cin>>map[i][j];
        }
    }
    for(int i=0;i<K;i++){
        cin>>a;
        v.push_back(a);
    }
    for(int i=0;i<K;i++){  // 굴러가는 과정
       
        int dx = x + dir[v[i]-1][0];
        int dy = y + dir[v[i]-1][1];
        if(0>dx || dx>=N || 0>dy || dy>=M) continue;  // 바깥으로 이동하려 하는 경우
        x = dx; // 주사위 위치 이동
        y = dy;
        move(v[i]); // 주사위 위치 이동
        if(map[x][y]==0){
            map[x][y]=dice[6];
        }
        else{
            dice[6]=map[x][y]; 
            map[x][y]=0;
        }
        cout<<dice[1]<<"\n"; // 윗 면 
    }
    
}
```
